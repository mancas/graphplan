(setq cena (make-lit 'cena nil))
(setq basura (make-lit 'basura T))
(setq manos (make-lit 'manosL T))
(setq preCocinar (list manos))
(setq preSacarBasura (list basura))
(setq effCocinar (list cena (not-obj manos)))
(setq effSacarBasura (list (not-obj basura) (not-obj manos)))
(setq cocinar (make-action 'cocinar preCocinar effCocinar))
(setq sacarBasura (make-action 'sacarBasura preSacarBasura effSacarBasura))
